.MODEL SMALL
.STACK 200H
.DATA
STR1  DB "MADAM$"
STR2 DB SIZEOF STR1 DUP(?)
MSG1 DB "YES! IT IS A PALINDROME$"
MSG2 DB "NO ! IT IS NOT A PALINDROME$"
.CODE
MAIN PROC
MOV AX,@DATA
MOV DS,AX

MOV CX,LENGTHOF STR1
SUB CX,1
MOV SI,LENGTHOF STR1
SUB SI,2
MOV DI,0

L1:
	MOV DL,STR1[SI]
	MOV STR2[DI],DL
	INC DI
	DEC SI
LOOP L1

MOV DI,0

L2:
	MOV AL, STR1[DI]
	CMP STR2[DI],AL
	JNE NOTPALINDROME
LOOP L2

MOV AH,9
MOV DX,OFFSET MSG1
INT 21H
JMP ENDPROGRAM

NOTPALINDROME:
MOV AH,9
MOV DX,OFFSET MSG2
INT 21H

ENDPROGRAM:
MOV AH,4CH
INT 21H
MAIN ENDP
END MAIN
